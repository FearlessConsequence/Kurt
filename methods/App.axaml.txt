public async Task<List<Stuff>> LoadStuff()
    {
        var list = new List<Stuff>();
        using var conn = new NpgsqlConnection(_connectionString);
        await conn.OpenAsync();
        string sql = @"               ";
        using var cmd = new NpgsqlCommand(sql, conn);
        using var reader = await cmd.ExecuteReaderAsync();
        while (await reader.ReadAsync())
        {
            list.Add(new Stuff
            {
               Id = reader.GetInt32(0),
               LastName = reader.GetString(1),
               FirstName = reader.GetString(2),
               Patronymic = reader.IsDBNull(3) ? "" : reader.GetString(3), 
               PhoneNumber = reader.GetString(4)
            });
        }
        
        return list;
    }


// selectttt authenticate
public async Task<Users> AuthenticateUser(string login, string password)
    {
        using var conn = new NpgsqlConnection(_connectionString);
        await conn.OpenAsync();

        string sql = @"            ";
        
        using var cmd = new NpgsqlCommand(sql, conn);
        cmd.Parameters.AddWithValue("@login", login);
        cmd.Parameters.AddWithValue("@password", password);
        using var reader = await cmd.ExecuteReaderAsync();

        if (await reader.ReadAsync())
        {
            return new Users
            {
                Id = reader.GetInt32(0),
                Login = reader.GetString(1),
                Password = reader.GetString(2)
            };
        }
        return null;
    }







// updateeee
public async Task UpdateStuff(Stuff stuff)
    {
        using var conn = new NpgsqlConnection(_connectionString);
        await conn.OpenAsync();
        
        string sql = @"            ";
        
        using var cmd = new NpgsqlCommand(sql, conn);
        cmd.Parameters.AddWithValue("@id", stuff.Id);
        cmd.Parameters.AddWithValue("@last_name", stuff.LastName);
        cmd.Parameters.AddWithValue("@first_name", stuff.FirstName);
        cmd.Parameters.AddWithValue("@patronymic", stuff.Patronymic);
        cmd.Parameters.AddWithValue("@phone_number", stuff.PhoneNumber);
        await cmd.ExecuteNonQueryAsync();
    }
